Функция Печать() Экспорт
	
	х = ЭтотОбъект.СсылкаНаОбъект;
   	ТабДокумент = Новый ТабличныйДокумент;
	
	
	ТипЗначения = ТипЗнч(х);
	ОбъектМетаданных = Метаданные.НайтиПоТипу(ТипЗначения);
	Если Метаданные.Справочники.Содержит(ОбъектМетаданных) Тогда
		ТабДокумент.Область("R3C1").Текст = "Справочники."+ТипЗначения;
		ТабДокумент.Область("R1C1").Текст = х.Код+" "+х.Наименование+" " +Строка(х.УникальныйИдентификатор());
	ИначеЕсли Метаданные.Документы.Содержит(ОбъектМетаданных) Тогда
		ТабДокумент.Область("R3C1").Текст = "Документы."+ТипЗначения;
		ТабДокумент.Область("R1C1").Текст = х.Номер+" "+х.Дата+" " +Строка(х.УникальныйИдентификатор());
	КонецЕсли;	
	
	ТабДокумент.Область("R2C1").Текст = ЗначениеВСтрокуВнутр(х);

	Возврат ТабДокумент;
